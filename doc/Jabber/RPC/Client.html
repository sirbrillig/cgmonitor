def do_rpc(xmlrpc)
  iq = Iq.new(:set, @jid)
  iq.from = @my_jid
  iq.id = IdGenerator::generate_id
  rpcquery = iq.add(IqQueryRPC.new)
  rpcquery.typed_add(xmlrpc)

  result = nil
  @stream.send_with_id(iq) do |iqreply|
    if iqreply.query.kind_of?(IqQueryRPC)
      result = iqreply.query.to_s
    end
  end

  result
end
